AMBIENCE - an integer
USING DATA in Sketch.js

function gotData(data) {
  var d = JSON.parse(data);
  proximity = d.proximity;
  color = d.color;
  ambience = d.ambience;
  gesture = d.gesture;
  console.log(
    "COLOR:" +
      JSON.stringify(color) +
      "  PROXIMITY:" +
      proximity +
      "   AMBIENCE:" +
      ambience +
      "  GESTURE:" +
      gesture
  );
}
